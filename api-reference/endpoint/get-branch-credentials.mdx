---
title: 'Get Branch Credentials'
openapi: 'GET /organizations/{organizationID}/projects/{projectID}/branches/{branchID}/credentials'
---

Retrieve branch credentials.

<ParamField path="organizationID" type="string" required>
  Unique identifier of the organization containing the project
</ParamField>

<ParamField path="projectID" type="string" required>
  Unique identifier of the project containing the branch
</ParamField>

<ParamField path="branchID" type="string" required>
  Unique identifier of the branch to retrieve credentials for
</ParamField>

<ParamField query="username" type="string">
  Username that the credentials requested for
</ParamField>

<ResponseField name="username" type="string">
  Username for accessing the branch database
</ResponseField>
<ResponseField name="password" type="string">
  Password for accessing the branch database
</ResponseField>
